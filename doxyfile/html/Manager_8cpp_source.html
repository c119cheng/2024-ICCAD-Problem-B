298 void Manager::assignSlot(FF* newFF){
299  int bit = newFF->getCell()->getBits();
300  if(bit == 1)
301  return ;
302 
303  vector<FF*> FFs = newFF->getClusterFF();
304  // Stable Marriage Problem (Gale-Shapley Algorithm)
305  std::vector<std::vector<double>> cost(bit, std::vector<double>(bit, 0)); // ith FF cost for putting it in j slot
306  std::queue<int> waitSlot; // slot wait to be assigned FF
307  std::priority_queue<std::pair<double, int>> pq[bit];
308  for(int i=0;i<bit;i++){
309  FF* curFF = FFs[i];
310  for(int j=0;j<bit;j++){
311  PrevInstance prevInstance = curFF->getPrevInstance();
312  Coor newCoorD = newFF->getNewCoor() + newFF->getPinCoor("D" + std::to_string(j));
313  double delta_hpwl = 0;
314  Coor inputCoor;
315  // D pin cost
316  if(prevInstance.instance){
317  if(prevInstance.cellType == CellType::IO){
318  inputCoor = prevInstance.instance->getCoor();
319  double old_hpwl = HPWL(inputCoor, curFF->getOriginalD());
320  double new_hpwl = HPWL(inputCoor, newCoorD);
321  delta_hpwl += old_hpwl - new_hpwl;
322  }
323  else if(prevInstance.cellType == CellType::GATE){
324  inputCoor = prevInstance.instance->getCoor() + prevInstance.instance->getPinCoor(prevInstance.pinName);
325  double old_hpwl = HPWL(inputCoor, curFF->getOriginalD());
326  double new_hpwl = HPWL(inputCoor, newCoorD);
327  delta_hpwl += old_hpwl - new_hpwl;
328  }
329  else{
330  FF* inputFF = dynamic_cast<FF*>(prevInstance.instance);
331  inputCoor = inputFF->getOriginalQ();
332  Coor newCoorQ = inputFF->getPhysicalFF()->getNewCoor() + inputFF->getPhysicalFF()->getPinCoor("Q" + inputFF->getPhysicalPinName());
333  double old_hpwl = HPWL(inputCoor, curFF->getOriginalD());
334  double new_hpwl = HPWL(newCoorQ, newCoorD);
335  delta_hpwl += old_hpwl - new_hpwl;
336  }
337  }
338  double newSlack = curFF->getTimingSlack("D") + DisplacementDelay * delta_hpwl;
339  cost[i][j] = newSlack < 0 ? newSlack : 0;
340 
341  // Q pin cost
342  for(auto& nextFF : curFF->getNextStage()){
343  Coor originalInput = curFF->getOriginalQ();
344  Coor newInput = newFF->getNewCoor() + newFF->getPinCoor("Q" + std::to_string(j));
345  if(nextFF.outputGate){
346  inputCoor = nextFF.outputGate->getCoor() + nextFF.outputGate->getPinCoor(nextFF.pinName);
347  double old_hpwl = HPWL(inputCoor, originalInput);
348  double new_hpwl = HPWL(inputCoor, newInput);
349  delta_hpwl = old_hpwl - new_hpwl;
350  }
351  else{
352  Coor newCoorD;
353  newCoorD = nextFF.ff->getPhysicalFF()->getNewCoor() + nextFF.ff->getPhysicalFF()->getPinCoor("D" + nextFF.ff->getPhysicalPinName());
354  double old_hpwl = HPWL(nextFF.ff->getOriginalD(), originalInput);
355  double new_hpwl = HPWL(newCoorD, newInput);
356  delta_hpwl = old_hpwl - new_hpwl;
357  }
358  newSlack = nextFF.ff->getTimingSlack("D") + DisplacementDelay * delta_hpwl;
359  cost[i][j] += newSlack < 0 ? newSlack : 0;
360  }
361  pq[j].push({cost[i][j], i});
362  }
363  waitSlot.push(i);
364  }
365 
366  for(int i=0;i<bit;i++)
367  FFs[i]->setPhysicalFF(nullptr, -1);
368 
369  while(!waitSlot.empty()){
370  int slot = waitSlot.front();
371  int preferFF = pq[slot].top().second; // this is idx = =
372  pq[slot].pop();
373  FF* curFF = FFs[preferFF];
374  if(curFF->getPhysicalFF() != newFF){ // is a single FF
375  curFF->setPhysicalFF(newFF, slot);
376  newFF->addClusterFF(curFF, slot);
377  waitSlot.pop();
378  }
379  else if(cost[preferFF][slot] < cost[preferFF][curFF->getSlot()]){ // is married, you want to be small three !!!!
380  // the one who is not loved is small three ~~~~~
381  waitSlot.push(curFF->getSlot()); // get find your new FF
382  curFF->setPhysicalFF(newFF, slot);
383  newFF->addClusterFF(curFF, slot);
384  waitSlot.pop();
385  }
386  }

492 double Manager::getOverallCost(bool verbose){
493  #ifdef NDEBUG
494  return 0;
495  #endif
496  double TNS_cost = 0;
497  double Power_cost = 0;
498  double Area_cost = 0;
499  double Bin_cost = 0;
500  for(const auto & ff_pair : FF_Map){
501  double curTNS = ff_pair.second->getTNS();
502  TNS_cost += alpha * (curTNS);
503  Power_cost += beta * ff_pair.second->getCell()->getGatePower();
504  Area_cost += gamma * (ff_pair.second->getCell()->getArea());
505  }
506 
507  // check for the bin density
508  if(!SKIP_BIN_CALC){
509  int numBins = 0;
510  int numViolationBins = 0;
511  int DieStartX = die.getDieOrigin().x;
512  int DieStartY = die.getDieOrigin().y;
513  int DieEndX = die.getDieBorder().x;
514  int DieEndY = die.getDieBorder().y;
515  int BinW = die.getBinWidth();
516  int BinH = die.getBinHeight();
517  #pragma omp parallel for reduction(+:numBins, numViolationBins) collapse(2)
518  for(int _x = DieStartX; _x < DieEndX; _x += BinW){
519  for(int _y = DieStartY; _y < DieEndY; _y += BinH){
520  numBins++;
521  double area = 0;
522  for(const auto &ff : FF_Map){
523  if(IsOverlap(Coor(_x, _y), die.getBinWidth(), die.getBinHeight(), ff.second->getNewCoor(), ff.second->getW(), ff.second->getH())){
524  area += ff.second->getW() * ff.second->getH();
525  }
526  }
527 
528  for(const auto &gate : Gate_Map){
529  if(IsOverlap(Coor(_x, _y), die.getBinWidth(), die.getBinHeight(), gate.second->getCoor(), gate.second->getW(), gate.second->getH())){
530  area += gate.second->getW() * gate.second->getH();
531  }
532  }
533 
534  // check if over bin max util
535  if(area / (die.getBinWidth() * die.getBinHeight()) > die.getBinMaxUtil()){
536  numViolationBins++;
537  }
538  }
539  }
540  Bin_cost = lambda * numViolationBins;
541  }
542 
543  double cost = TNS_cost + Power_cost + Area_cost + Bin_cost;
544  double TNS_percentage = TNS_cost / cost * 100;
545  double Power_percentage = Power_cost / cost * 100;
546  double Area_percentage = Area_cost / cost * 100;
547  double Bin_percentage = Bin_cost / cost * 100;
548  if(verbose){
549  if(SKIP_BIN_CALC) std::cout << "[Warning] Skip bin density calculation!" << std::endl;
550  size_t numAfterDot = 4;
551  std::vector<std::string> header = {"Cost", "Weight", "Value", "Percentage(%)"};
552  std::vector<std::vector<std::string>> rows = {
553  {"TNS", toStringWithPrecision(alpha, numAfterDot), toStringWithPrecision(TNS_cost, numAfterDot), toStringWithPrecision(TNS_percentage, numAfterDot) + "(%)"},
554  {"Power", toStringWithPrecision(beta, numAfterDot), toStringWithPrecision(Power_cost, numAfterDot), toStringWithPrecision(Power_percentage, numAfterDot) + "(%)"},
555  {"Area", toStringWithPrecision(gamma, numAfterDot), toStringWithPrecision(Area_cost, numAfterDot), toStringWithPrecision(Area_percentage, numAfterDot) + "(%)"},
556  {"Bin", toStringWithPrecision(lambda, numAfterDot), toStringWithPrecision(Bin_cost, numAfterDot), toStringWithPrecision(Bin_percentage, numAfterDot) + "(%)"},
557  {"Total", "-", toStringWithPrecision(cost, numAfterDot), "100.00(%)"},
558  {"WNS", toStringWithPrecision(getWNS(), numAfterDot), "TNS", toStringWithPrecision(getTNS(), numAfterDot)}
559  };
560 
561  PrettyTable pt;
562  pt.AddHeader(header);
563  pt.AddRows(rows);
