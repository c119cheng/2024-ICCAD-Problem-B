441 double Legalizer::PlaceFF(Node *ff, size_t row_idx, bool &placeable){
442  double minDisplacement = ff->getDisplacement();
443  const auto &subrows = rows[row_idx]->getSubrows();
444  bool skip = false;
445  bool rowCanPlace = false;
446 
447  for(size_t i = 0; i < subrows.size(); i++){
448  const auto &subrow = subrows[i];
449  if(subrow->hasCell(ff->getCell())) continue;
450  double alignedStartX = rows[row_idx]->getStartCoor().x + std::ceil((int)(subrow->getStartX() - rows[row_idx]->getStartCoor().x) / rows[row_idx]->getSiteWidth()) * rows[row_idx]->getSiteWidth();
451  bool subrowSkip = false;
452  bool subrowCanPlace = false;
453  for(int x = alignedStartX; x <= subrow->getEndX(); x += rows[row_idx]->getSiteWidth()){
454  Coor currCoor = Coor(x, rows[row_idx]->getStartCoor().y);
455  if(ff->getDisplacement(currCoor) > minDisplacement){
456  subrowSkip = true;
457  skip = true;
458  Coor subrowEndCoor = Coor(subrow->getEndX(), rows[row_idx]->getStartCoor().y);
459  // If current subrow can't find better solution
460  if(ff->getDisplacement(subrowEndCoor) > minDisplacement) break;
461  continue;
462  }
463  bool canPlace = ContinousAndEmpty(x, rows[row_idx]->getStartCoor().y, ff->getW(), ff->getH(), row_idx);
464  double displacement = ff->getDisplacement(currCoor);
465 
466  if(canPlace){
467  rowCanPlace = true;
468  subrowCanPlace = true;
469  if(displacement < minDisplacement){
470  minDisplacement = displacement;
471  ff->setLGCoor(currCoor);
472  ff->setIsPlace(true);
473  ff->setPlaceRowIdx(row_idx);
474  }
475  }
476  }
477  if(!subrowCanPlace && !subrowSkip)
478  subrow->addRejectCell(ff->getCell());
479 
480  }
481  if(!skip && !rowCanPlace){
482  placeable = false;
483  }
484  return minDisplacement;
485 }

488 bool Legalizer::ContinousAndEmpty(double startX, double startY, double w, double h, int row_idx){
489  double endY = startY + h;
490  double currentY = startY;
491 
492  for(size_t i = row_idx; i < rows.size(); i++){
493  double rowStartY = rows[i]->getStartCoor().y;
494  double rowStartX = rows[i]->getStartCoor().x;
495  double rowEndX = rowStartX + rows[i]->getSiteWidth() * rows[i]->getNumOfSite();
496 
497  // check if the current row is upper than currentY -> early break
498  if(rowStartY > currentY) break;
499  // check if the current row is target row -> jump condition
500  if(rowStartY != currentY || startX < rows[i]->getStartCoor().x || startX + w > rowEndX) continue;
501  // Save highest space can place in range startX to startX + w
502  double placeH = DBL_MAX;
503  // check if this row is continuous from startX to startX + w
504  if(rows[i]->canPlace(startX, startX + w, placeH)){
505  // update current continuous y
