377 void Gradient::Step(bool onlyNegative) {
378  // Compute the gradient direction
379  obj_.forward();
380  obj_.backward(step_, onlyNegative);
381  // Compute the Polak-Ribiere coefficient and conjugate directions
382  double beta; // Polak-Ribiere coefficient
383  if (step_ == 0) {
384  // For the first step, we will set beta = 0 and d_0 = -g_0
385  beta = 0.;
386  for (size_t i = 0; i < kNumModule; ++i) {
387  dir[i].x = -obj_.grad().at(i).x;
388  dir[i].y = -obj_.grad().at(i).y;
389  }
390  } else {
391  // For the remaining steps, we will calculate the Polak-Ribiere coefficient and
392  // conjugate directions normally
393  double t1 = 0.; // Store the numerator of beta
394  double t2 = 0.; // Store the denominator of beta
395  #pragma omp parallel for reduction(+:t1, t2)
396  for (size_t i = 0; i < kNumModule; ++i) {
397  Coor g = obj_.grad().at(i);
398  Coor t3;
399  t3.x = g.x * (g.x - grad_prev_.at(i).x);
400  t3.y = g.y * (g.y - grad_prev_.at(i).y);
401  t1 += t3.x + t3.y;
402  t2 += std::abs(g.x) + std::abs(g.y);
403  }
404  beta = t1 / (t2 * t2 + 0.00001);
405  #pragma omp parallel for num_threads(MAX_THREADS)
406  for (size_t i = 0; i < kNumModule; ++i) {
407  dir[i].x = -obj_.grad().at(i).x + beta * dir_prev_.at(i).x;
408  dir[i].y = -obj_.grad().at(i).y + beta * dir_prev_.at(i).y;
409  }
410  }
411  // Update the solution
412  #pragma omp parallel for num_threads(MAX_THREADS)
413  for(size_t idx=0;idx<kNumModule;idx++){
414  // FF* ff = ff_m.second;
415  FF* ff = FFs[idx];
416  Coor coor = ff->getCoor();
417  coor.x = coor.x + alpha_ * dir[idx].x;
418  coor.y = coor.y + alpha_ * dir[idx].y;
419  ff->setCoor(coor);
420  ff->setNewCoor(coor);
421  }
422 
423  // Update the cache data members
424  grad_prev_ = obj_.grad();
425  dir_prev_ = dir;
426  step_++;
427 }
