214 double FF::shift(std::vector<FF *> &FFs){
215  double x_shift = 0;
216  double y_shift = 0;
217  double scale_factor = 0;
218  for(size_t i = 0; i < NeighborFFs.size(); i++){
219  FF *ffneighbor = FFs[NeighborFFs[i].first];
220  double bandwidth_i = ffneighbor->getBandwidth();
221  double weight = GaussianKernel(newCoor, ffneighbor->getCoor(), bandwidth_i) / std::pow(bandwidth_i, 4);
222  // DEBUG_MSG(weight)
223  x_shift += ffneighbor->getCoor().x * weight;
224  y_shift += ffneighbor->getCoor().y * weight;
225  scale_factor += weight;
226  }
227  assert(std::isnormal(scale_factor));
228  x_shift /= scale_factor;
229  y_shift /= scale_factor;
230  double euclidean_distance = std::sqrt(SquareEuclideanDistance({x_shift, y_shift}, newCoor));
231  setNewCoor({x_shift, y_shift});
232  return euclidean_distance;
233 }
