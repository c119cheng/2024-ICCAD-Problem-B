510  for(size_t i=0;i<FFs.size();i++){
511  // for(auto& ff_m : FF_list){

524  delta_hpwl += old_hpwl - new_hpwl;

529  double new_hpwl = HPWL(inputCoor, cur_ff->getCoor() + cur_ff->getPinCoor("D"));
530  delta_hpwl += old_hpwl - new_hpwl;
531  }
532  else{

534  inputCoor = prevFF->getOriginalQ();

536  double new_hpwl = HPWL(prevFF->getCoor() + prevFF->getPinCoor("Q"), cur_ff->getCoor() + cur_ff->getPinCoor("D"));
537  delta_hpwl += old_hpwl - new_hpwl;
538  }
539  }
